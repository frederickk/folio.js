117 	 *
118 	 *	http://www.siafoo.net/snippet/191
119 	 *
120 	 *	Returns a number between v1 and v2, including v1 but not v2.
121 	 *	The bias represents the preference towards lower or higher numbers,
122 	 *	as a number between 0.0 and 1.0. For example: 
123 	 *	random(0, 10, bias=0.9) will return 9 much more often than 1.
124 	 *
125 	 */
126 	randomBias : function(minr, maxr, bias) {
127 		var _map = new Array(90.0, 9.00, 4.00, 2.33, 1.50, 1.00, 0.66, 0.43, 0.25, 0.11, 0.01);
128 		bias = Math.max(0, Math.min(bias, 1)) * 10;
129 
130 		var i = parseInt(Math.floor(bias))
131 		var n = _map[i]
132 		if(bias < 10) n += (_map[i+1]-n) * (bias-i);
133 
134 		return Math.pow( Math.random(),n ) * (maxr-minr) + minr;
135 	},

493 	 *			0.0 - 1.0
494 	 *	@param v

501 	this.HSVtoColor = function(h, s, v) {
502 		var r, g, b;
503 		var RGB = new paper.RGBColor(Math.round(v), Math.round(v), Math.round(v));
504 		if(s!=0) { 
505 			// h must be < 1
506 			var var_h = h * 6;
507 			if (var_h==6) var_h = 0;
508 			var var_i = Math.floor( var_h );
509 			var var_1 = v*(1-s);
510 			var var_2 = v*(1-s*(var_h-var_i));
511 			var var_3 = v*(1-s*(1-(var_h-var_i)));
512 			if(var_i==0) { 
513 				var_r = v; 
514 				var_g = var_3; 
515 				var_b = var_1;
516 			}
517 			else if(var_i==1) { 
518 				var_r = var_2;
519 				var_g = v;
520 				var_b = var_1;
521 			}
522 			else if(var_i==2) { 
523 				var_r = var_1;
524 				var_g = v;
525 				var_b = var_3
526 			}
527 			else if(var_i==3) { 
528 				var_r = var_1;
529 				var_g = var_2;
530 				var_b = v;
531 			}
532 			else if (var_i==4) { 
533 				var_r = var_3;
534 				var_g = var_1;
535 				var_b = v;
536 			}
537 			else { 
538 				var_r = v;
539 				var_g = var_1;
540 				var_b = var_2
541 			}
542 			RGB.red = Math.round(var_r);
543 			RGB.green = Math.round(var_g);
544 			RGB.blue = Math.round(var_b);
545 		}

618 Frederickk.FFade = function() {
619 	// Properties
620 	// private
621 	var fadeMillis = 1000; // set to default of 1s OR 1000ms
622 	
623 	var timeStartFade = 0.0;
624 	var timeEndFade = 0.0;
625 	
626 	var bBeginFade = false;
627 	var bFadeIn = false;
628 	var bFadeOut = false;
629 
630 	// public
631 	this.alpha = 1.0;
632 
633 
634 	
635 	// Methods
636 	this.update = function() {
637 		_currentTime = new Date().getMilliseconds();
638 		this.update(_currentTime);
639 	};
640 	this.update = function(currentTime) {
641 		if(bBeginFade) {
642 			bBeginFade = false;
643 			timeStartFade = currentTime;
644 			if(bFadeIn) {
645 				timeEndFade = currentTime + parseFloat((1.0 - this.alpha) * fadeMillis);
646 			}
647 			else {
648 				timeEndFade = currentTime + parseFloat(this.alpha*fadeMillis);
649 			}
650 			if(timeEndFade == currentTime) {
651 				if(bFadeIn) {
652 					bFadeIn = false;
653 					this.alpha = 1.0;
654 				}
655 				else {
656 					bFadeOut = false;
657 					this.alpha = 0.0;
658 				}
659 			}
660 		}
661 		if(bFadeIn) {
662 			this.alpha = 1.0 - parseFloat((timeEndFade - currentTime) / fadeMillis);
663 			if(currentTime > timeEndFade) {
664 				bFadeIn = false;
665 				this.alpha = 1.0;
666 			}
667 		}
668 		else if(bFadeOut) {
669 			this.alpha = parseFloat((timeEndFade - currentTime) / fadeMillis);
670 			if(currentTime > timeEndFade) {
671 				bFadeIn = false;
672 				this.alpha = 0.0;
673 			}
674 		}
675 	};
676 
677 	this.fadeIn = function() {
678 		if(bFadeIn)return;
679 		if(this.alpha == 1.0) return;
680 		bBeginFade = true;
681 		bFadeIn = true;
682 		bFadeOut = false;
683 	};
684 	this.fadeOut = function() {
685 		if(bFadeOut)return;
686 		if(this.alpha == 0.0) return;
687 		bBeginFade = true;
688 		bFadeOut = true;
689 		bFadeIn = false;
690 	};
691 
692 	this.isFadingIn = function() {
693 		return bFadeIn;
694 	};
695 	this.isFadingOut = function() {
696 		return bFadeOut;
697 	};
698 	this.stopFade = function() {
699 		bBeginFade = bFadeIn = bFadeOut = false;
700 	};
